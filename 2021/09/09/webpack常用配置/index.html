webpack常用配置

发表于 2021-09-09 更新于 2021-09-12 阅读次数：

webpack

webpack:一种前端资源构建工具，一个静态模块打包器

webpack本身可以打包js/json文件，不能打包样式等其他资源，这就需要自己去配置webpack.config.js(配置文件),且在终端输入webpack，会加载里面的配置去打包文件。所有的构建工具都是基于nodejs平台运行的，模块化默认采用commonjs.

webpack开发环境的基本配置

//resolve用来拼接绝对路径的方法
const {resolve} = require('path');
//引入插件
const HtmlWebpackPlugin = require('./html-webpack-plugin');
module.exports = {
//webpack配置
//打包入口文件
entry: './src/index.js,
//输出文件
output:{
    //输出文件名
  filename:'bulit.js',
    //输出路径
    //_dirname nodejs的变量，代表当前文件的目录的绝对路径
  path: resolve(_dirname,'build')
    },
//loader的配置
module:{
    //详细的loader配置
    rules:{
        test:/\.less$/,
        use:{
            'style-loader',
            'css-loader',
            'less-loader'
        }
    },
    rules:{
            //匹配哪些文件
            test:/\.css$/,
            //使用哪些loader进行处理
            use:[
                //use数组中loader执行顺序：从右到左，从下到上，依次执行
                //创建style标签，将js中的样式资源插入到html文件里的head标签中生效
                'style.loader',
                //将css文件变成commonjs模块加载到js中，里面内容是样式字符串
                'css-loader'
            ]
        },
         rules:{
            test:/\.(jpg|png|gif)$/,
            options:{
                limit: 8*1024,//不能大于8KB
                name:'[hash:10].[ext]'//文件压缩后的名字
                outputPath: 'imgs',
                //url-loader默认使用es6模块化解析，而html-loader引入图片是commonjs，解析时会出问题
                esModule:false
            }
        },
        {
            //处理其他资源
            exclude:/\.(html|js|css|less|jpg|png|gif)/,
            loader:'file-loader',
            options:{
                name:'[hash:10].[ext]'//文件压缩后的名字
            }
        }
    }
//plugins的配置
plugins:{
    //详细plugins的配置
    new HtmlWebpackPlugin({
        template:'./src/index.html'
    })
},
devServer:{
    contentBase:resolve(_dirname,'bulid'),
    compress:true,
    port:3000,
    open:true
}
//模式
mode:'development',//开发模式
//mode:'production' //生产模式
}

提取css成单独文件

//详细的loader配置
    rules:{
            //匹配哪些文件
            test:/\.css$/,
            //使用哪些loader进行处理
            use:[
                //use数组中loader执行顺序：从右到左，从下到上，依次执行
                //作用：提取js中的css成单独文件,z这个loader取代style-loader。
                MiniCssExtractPlugin.loader,
                //将css文件整合到js文件中
                'css-loader'
                {
                //对css做兼容处理
                //还需要在package.json中定义browserslist
                    loader: 'postcss-loader',
                    options:{
                        ident: 'postcss',
                        plugins:()=>{
                            require('_postcss-preset-evn')()
                        }
                    }
                }
            ]
        },

webpack生产环境的基本配置

//resolve用来拼接绝对路径的方法
const {resolve} = require('path');
//引入插件
const HtmlWebpackPlugin = require('./html-webpack-plugin');
//提取css成单独文件的插件
const MiniCssExtractPlugin = require('mini-css-extract-plugin');
//对js做压缩的插件
const OptimizeCssAssetsWebpackPlugin = require('optimize-css-assets-webpack-plugin');
module.exports = {
//webpack配置
//打包入口文件
entry: './src/index.js,
//输出文件
output:{
    //输出文件名
  filename:'bulit.js',
    //输出路径
    //_dirname nodejs的变量，代表当前文件的目录的绝对路径
  path: resolve(_dirname,'build')
    },
//loader的配置
module:{
    //详细的loader配置
    rules:{
            //匹配哪些文件
            test:/\.css$/,
            //使用哪些loader进行处理
            use:[
                //use数组中loader执行顺序：从右到左，从下到上，依次执行
                //作用：提取js中的css成单独文件,z这个loader取代style-loader。
                MiniCssExtractPlugin.loader,
                //将css文件整合到js文件中
                'css-loader'
                {
                //对css做兼容处理
                //还需要在package.json中定义browserslist
                    loader: 'postcss-loader',
                    options:{
                        ident: 'postcss',
                        plugins:()=>{
                            require('_postcss-preset-evn')()
                        }
                    }
                }
            ]
        },
    rules:{
        test:/\.less$/,
        use:{
            MiniCssExtractPlugin.loader,
            'css-loader',
            {
                //对css做兼容处理
                //还需要在package.json中定义browserslist
                    loader: 'postcss-loader',
                    options:{
                        ident: 'postcss',
                        plugins:()=>{
                            require('_postcss-preset-evn')()
                        }
                    }
                }
            'less-loader'
        }
    },
     rules:{
        test:/\.(jpg|png|gif)$/,
        options:{
            limit: 8*1024,//不能大于8KB
            name:'[hash:10].[ext]'//文件压缩后的名字,
            outputPath: 'imgs',
            //url-loader默认使用es6模块化解析，而html-loader引入图片是commonjs，解析时会出问题
            esModule:false

        }
    },
    {
        //处理其他资源
        exclude:/\.(html|js|css|less|jpg|png|gif)/,
        loader:'file-loader',
        options:{
            name:'[hash:10].[ext]'//文件压缩后的名字
        }
    },
    //对js做语法检查
    {
        //还需在package.json中eslintConfig 设置成 airbnb
        test: /\.js$/,
        exclude: /node_modules/,
        loader: 'eslint-loader',
        //自动修改js中的语法错误
        options:{
            fix:true
        }
    },
    //对js做兼容性处理
    {
        //还需在package.json中eslintConfig 设置成 airbnb
        test: /\.js$/,
        exclude: /node_modules/,
        loader: 'babel-loader',
        options:{
            presets:{
                '@babel/preset-env',
                {
                    useBuiltIns: 'usage',
                    corejs:{version:3},
                    targets:{
                        chrome: '60',
                        firefox: '50'
                    }
                }
            }
        }
    },
    //html处理不了图片，所有要用html-loader
    {
        test: /\.html$/,
        loader: 'html-loader'
    }
    }
//plugins的配置
plugins:{
    //详细plugins的配置
    new HtmlWebpackPlugin({
        template:'./src/index.html',
        //对html做压缩处理
        minify:{
            collapseWhitespace:true,//去除空格
            removeComments:true//去除注释
        }
    })，
    new MiniCssExtractPlugin({
        //重命名css文件
        filename: 'css/built.css'
    }),
    //对js做压缩处理
    new OptimizeCssAssetsWebpackPlugin()
},
devServer:{
    contentBase:resolve(_dirname,'bulid'),
    compress:true,
    port:3000,
    open:true
}
//模式
//mode:'development',//开发模式
mode:'production' //生产模式
}
